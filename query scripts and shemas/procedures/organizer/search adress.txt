
--поиск адреса вместе с домом
CREATE OR REPLACE FUNCTION searchAddress(word_ text, house_ text) 
RETURNS table (j json) AS
$$
BEGIN
    RETURN QUERY  SELECT to_json(a) FROM (
		select id,street,house from address where deleted = false 
		and street ~* word_ and house ~* house_ LIMIT 15)a;
END;
$$ LANGUAGE plpgsql;


--поиск адреса
CREATE OR REPLACE FUNCTION searchAddress(word_ text) 
RETURNS table (j json) AS
$$
BEGIN
    RETURN QUERY  SELECT to_json(a) FROM (select id,street,house from address where deleted = false and (street ~* word_ or house ~* word_) LIMIT 15)a;
END;
$$ LANGUAGE plpgsql;

--поиск адреса
CREATE OR REPLACE FUNCTION searchAddress(word_ text) 
RETURNS table (j json) AS
$$
BEGIN
    RETURN QUERY  SELECT to_json(a) FROM (select id,street,house from address where street ilike word_ or house ilike word_ LIMIT 15)a;
END;
$$ LANGUAGE plpgsql;
